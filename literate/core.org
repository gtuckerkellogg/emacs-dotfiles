* Backups, autosave, recent files, places, history
#+begin_src emacs-lisp
(let ((backup-dir (gtk/emacs-path "backups/"))
      (auto-dir   (gtk/emacs-path "auto-save/")))
  (make-directory backup-dir t)
  (make-directory auto-dir t)
  (setq backup-directory-alist `(("." . ,backup-dir))
        auto-save-file-name-transforms `((".*" ,auto-dir t))
        backup-by-copying t
        delete-old-versions t
        version-control t))
(add-to-list 'completion-ignored-extensions ".los")
(recentf-mode 1)
(global-set-key (kbd "C-x f") #'recentf-open)
(save-place-mode 1)
(savehist-mode 1)
#+end_src

* Remote editing (tramp)
#+begin_src emacs-lisp
;; Edit remote files as root by hopping through ssh first
;; (e.g. /sudo:host:/etc/... connects via /ssh:host: then sudo).
(with-eval-after-load 'tramp
  (setq tramp-default-proxies-alist '((".*" "\\`root\\'" "/ssh:%h:"))))
#+end_src
